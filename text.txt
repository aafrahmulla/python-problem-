''Given an integer, , print the following values for each integer  i from 1  to n:
Decimal
Octal
Hexadecimal (capitalized)
Binary
#The four values must be printed on a single line in the order specified above for each  i from 1to n. Each value should be space-padded to match the width of the binary value of ''' 
 
def print_formatted(number):
    b=str(bin(n))[2:]
    padded_bin=len(b)
    for i in range (1,n+1):
        o=str(oct(i))[2:]
        h=str(hex(i))[2:]
        h=h.upper()
        b=str(bin(i))[2:]
        d=str(i)
        print(f'{d:>{padded_bin}} {o:>{padded_bin}} {h:>{padded_bin}} {b:>{padded_bin}}')
   

if __name__ == "__main__": 
    n=int(input())
    print_formatted(n)
